- 2.给一个组件传值

````
// 1.设置props值,并制定数据类型

	name: 'orglist',
	props: {
		data: Array
	}
}
// js引入这个组件
export default {
	data() {
		return {
			datas: []
		}
	},
	components: {
		OrgList
	}
}
// html使用这个组件,并传入所需要的数据
<org-list :data="datas"></org-list>
````
